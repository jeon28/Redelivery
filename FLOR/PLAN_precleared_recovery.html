3.1 A Status 입력 가시성 우회

async def _fill_unit_input(self, container: str) -> bool:
    """Unit Number 입력란에 값 주입. visible 실패 시 JS 직접 주입으로 대체."""
    sel = 'input[placeholder="Unit Number"]'
    # 1) attached 까지 대기 (탭 전환 직후 DOM 아직 안 붙은 경우)
    try:
        await self.page.wait_for_selector(sel, state="attached", timeout=10_000)
    except Exception:
        return False

    # 2) scrollIntoView 후 visible 시도 (짧게)
    try:
        loc = self.page.locator(sel).first
        await loc.scroll_into_view_if_needed(timeout=3_000)
        await loc.fill(container, timeout=5_000)
        return True
    except Exception:
        pass

    # 3) JS evaluate fallback — Vue/Element UI input 이벤트로 v-model 갱신
    ok = await self.page.evaluate(r"""(args) => {
        const el = document.querySelector(args.sel);
        if (!el) return false;
        el.focus();
        const setter = Object.getOwnPropertyDescriptor(
            window.HTMLInputElement.prototype, 'value').set;
        setter.call(el, args.val);
        el.dispatchEvent(new Event('input', { bubbles: true }));
        el.dispatchEvent(new Event('change', { bubbles: true }));
        return true;
    }""", {"sel": sel, "val": container})
    return bool(ok)

_search_one 의 await unit_input.fill(container) 호출을 await self._fill_unit_input(container) 로 교체.

const PRECLEARED_RE = /already\s+precleared|Redelivery\s+number:\s*PP[RF]\d+/i;
const UNIT_RE = /^[A-Z]{4}\d{6,7}$/;

// 1차 패스: cells 자체에 unit + reason 둘 다 있으면 기존 로직 그대로
// (단일 테이블 케이스 호환)
//   ... existing pass ...

// 2차 패스: 분리행 매칭
for (let i = 0; i < allRows.length; i++) {
    const cells = allRows[i].cells;
    const reason = cells.find(c => PRECLEARED_RE.test(c) || c.length >= 40);
    if (!reason) continue;
    if (cells.some(c => UNIT_RE.test(c))) continue;  // 1차에서 처리됨

    // 인접 ±3 rows 에서 unit-only 행 검색 (같은 시각적 행 후보)
    let found = null;
    for (const d of [0, 1, -1, 2, -2, 3, -3]) {
        const j = i + d;
        if (j < 0 || j >= allRows.length || j === i) continue;
        const u = allRows[j].cells.find(c => UNIT_RE.test(c));
        if (u) { found = u; break; }
    }
    if (found) {
        out.invalid.push({ unit: found, reason, cells });
    }
}

Order Date ≠ Expiry Date: Step3 텍스트엔 Order Date(신청일) 만 노출됨. UI 의 "유효기간" 열은 만료일(Expiry Date) 이므로 Order Date 를 매핑하지 않음. 유효기간은 Status 경로(_expand_and_extract)가 성공해야 채워짐. B 경로는 PPR 번호·Depot 만 회수하고 close_date=None 유지.
